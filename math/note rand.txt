package main

import (
	"crypto/rand"
	"fmt"
	"math/big"
)

func main() {
	// Sinh một số nguyên ngẫu nhiên an toàn trong khoảng [0, 100)
	randomInt, err := rand.Int(rand.Reader, big.NewInt(100))
	if err != nil {
		fmt.Println("Error generating random number:", err)
		return
	}

	fmt.Println("Random number:", randomInt)
}
